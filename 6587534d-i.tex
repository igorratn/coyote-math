\documentclass[tikz,border=6pt]{standalone}
\usepackage{amsmath}
\usetikzlibrary{arrows.meta}

\begin{document}
\begin{tikzpicture}[scale=6, >=Stealth]

% ---------- Params ----------
\def\P{(0.20,0.25)}
\def\Q{(0.80,0.75)}
% choose the shortest lift: Q' = Q + (-1,0)
\def\Qlift{(-0.20,0.75)}

% ---------- Tiling (3x3 light copies) ----------
\foreach \i in {-1,0,1}{
  \foreach \j in {-1,0,1}{
    \draw[very thin, gray!40] (\i,\j) rectangle ++(1,1);
  }
}

% ---------- Central fundamental square ----------
\draw[line width=0.5pt, black] (0,0) rectangle (1,1);

% Edge-identification arrows on central square
\draw[->, black!70] (-0.02,0.15) -- ++(0.12,0) ;
\draw[->, black!70] (1.02,0.15) -- ++(-0.12,0);
\draw[->, black!70] (0.15,-0.02) -- ++(0,0.12);
\draw[->, black!70] (0.15,1.02) -- ++(0,-0.12);

% ---------- Points P and Q ----------
\fill[black] \P circle (0.015) node[below left=2pt] {$P$};
\fill[black] \Q circle (0.015) node[above right=2pt] {$Q$};

% Also show the chosen lift Q' in the left neighbor cell
\fill[black!50] \Qlift circle (0.012) node[left=2pt, black!50] {$Q'$};

% ---------- Candidate paths ----------
% (A) Correct geodesic: straight segment from P to Q' (now dashed)
\draw[line width=1.1pt, blue, dashed]
  \P -- \Qlift;
\fill[blue] \Qlift circle (0.015); % reinforce end dot in blue
\node[blue] at (0.05,0.35) {\Large A}; % Below the path

% (B) Curved inside-square path (not a geodesic on flat torus)
\draw[line width=0.8pt, red!70, dashed]
  plot[smooth,tension=1] coordinates {(0.20,0.25) (0.35,0.85) (0.55,0.65) (0.80,0.75)};
\node[red!70] at (0.63,0.58) {\Large B};

% (C) Broken two-segment path (also not a geodesic)
\draw[line width=0.8pt, orange!80, dash dot]
  (0.20,0.25) -- (0.20,0.75) -- (0.80,0.75);
\node[orange!90] at (0.65,0.80) {\Large C}; % Above the path

% (D) Modified path: wraps with tilted end at Q' equivalent
\draw[line width=0.8pt, purple!70, dotted]
  (0.20,0.25) -- (1.20,0.25) -- (1.20,1.25) -- (-0.80,1.25) -- (-0.20,0.75);
\fill[purple!70] (-0.20,0.75) circle (0.015); % reinforce end dot in purple at Q'
\node[purple!70] at (0.88,0.20) {\Large D};

\end{tikzpicture}
\end{document}
